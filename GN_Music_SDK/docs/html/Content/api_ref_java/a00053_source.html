919  void postSetCoverArtSizePreference() {
920  // property must be made up of comma separated strings
921  // "SMALL, MEDIUM, THUMBNAIL, LARGE, XLARGE"
922 
923  boolean isInvalid = false;
924  int tokenCount = 0;
925 
926  StringBuilder builder = new StringBuilder();
927 
928  String coverArtSize = getProperty("content.coverart.sizepreference");
929 
930  StringTokenizer st = new StringTokenizer(coverArtSize, ",");
931 
932  while (st.hasMoreTokens()) {
933  String token = (String) st.nextToken();
934  token = token.trim();
935  if (token.equals("LARGE") ||
936  token.equals("XLARGE") ||
937  token.equals("SMALL") ||
938  token.equals("MEDIUM") ||
939  token.equals("THUMBNAIL")) {
940  builder.append(token);
941  builder.append(",");
942  tokenCount++;
943  } else {
944  isInvalid = true;
945  }
946  }
947  if (tokenCount == 0) {
948  isInvalid = true;
949  } else {
950  builder.setLength(builder.length() - 1);
951  }
952 
953  if (isInvalid) {
954  // Reset to default if attempt is made to set to an invalid size pref
955  this.publicProperties.put("content.coverart.sizepreference",defaultCoverArtSizePreference);
956  } else {
